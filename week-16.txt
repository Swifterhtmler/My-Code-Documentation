}

Tavalinen funktio:

function functionName(...args) {

}

Argumenttien määrittely parametreinä funktiossa:

function functionName(...args) {
    return function() {
        return undefined;
    };
}

const multiply = (a, b) => (a * b);
let result = multiply(1, 2);

    let result = a * a;
    return result;
}

function exponetent(a) {
  let result = a * a;
  return result;
